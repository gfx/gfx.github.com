177=head1 NAME
178
179Plack::Handler::CLI - Command line interface to PSGI applications
180
181=head1 VERSION
182
183This document describes Plack::Handler::CLI version 0.03.
184
185=head1 SYNOPSIS
186
187 #!perl -w
188 # a cat(1) implementation on PSGI/CLI
189 use strict;
190 use Plack::Handler::CLI;
191 use URI::Escape qw(uri_unescape);
192
193 sub err {
194 my(@msg) = @_;
195 return [
196 500,
197 [ 'Content-Type' => 'text/plain' ],
198 \@msg,
199 ];
200 }
201
202 sub main {
203 my($env) = @_;
204
205 my @files = split '/', $env->{PATH_INFO};
206
207 local $/;
208
209 my @contents;
210 if(@files) {
211 foreach my $file(@files) {
212 my $f = uri_unescape($file);
213 open my $fh, '<', $f
214 or return err("Cannot open '$f': $!\n");
215
216 push @contents, readline($fh);
217 }
218 }
219 else {
220 push @contents, readline($env->{'psgi.input'});
221 }
222
223 return [
224 200,
225 [ 'Content-Type' => 'text/plain'],
226 \@contents,
227 ];
228 }
229
230 my $handler = Plack::Handler::CLI->new(need_headers => 0);
231 $handler->run(\&main);
232
233=head1 DESCRIPTION
234
235Plack::Handler::CLI is a PSGI handler which provides a command line interface
236for PSGI applications.
237
238=head1 INTERFACE
239
240=head2 B<< Plack::Handler::CLI->new(%options) : CLI >>
241
242Creates a Plack handler that implements a command line interface.
243
244PSGI headers will be printed by default, but you can suppress them
245by C<< need_headers => 0 >>.
246
247=head2 B<< $cli->run(\&psgi_app, @argv) : Void >>
248
249Runs I<&psgi_app> with I<@argv>.
250
251C<< "--key" => "value" >> (or C<< "--key=value" >>) pairs in I<@argv> are
252packed into C<QUERY_STRING>, while any other arguments are packed into
253C<PATH_INFO>, so I<&psgi_app> can get command line arguments
254as PSGI parameters.
